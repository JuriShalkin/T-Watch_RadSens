Шрифты необходимо прописать в файле \Arduino\libraries\TTGO_TWatch_Library-master\src\TFT_eSPI\User_Setups\User_Custom_Fonts.h или \Arduino\libraries\TFT_eSPI\User_Setups\User_Custom_Fonts.h

#include "../Fonts/Custom/Orbitron_Light_32.h" // CF_OL32
